3.6.1.1. Source code#

  1// University of Florida EEL6528
  2// Tan F. Wong
  3// Feb 2, 2021
  4
  5#ifndef FFT_H
  6#define FFT_H
  7#include <complex>
  8#include <fftw3.h>
  9//#include <thread>
 10#ifdef USE_VOLK
 11#include <volk/volk.h>
 12#endif
 13
 14// 1-d, single-precision FFT class
 15// All samples in the arrays in, out are contiguous
 16// out_nsteps >= fftsize
 17// Use multi-threading in FFTW3
 18class fft{
 19  private:
 20    fftwf_plan plan; // FFT plan
 21    int nthreads;   // Number of threads
 22    int fftsize;    // FFT size
 23    int nblocks;    // Number of FFT blocks to calculate along a row
 24    int in_nsteps;  // How many samples to next block in array in
 25    int out_nsteps; // How many samples to next block in array out
 26    int U_D;          // # of rows in 2-d array = Up/Down-sampling factor
 27    bool inverse;   // FFT = false, IFFT = true
 28    std::complex<float>* in;      // Pointer to input array
 29    std::complex<float>* out;     // Pointer to output array
 30    std::complex<float>* ptr_in;  // Pointer to hold if want to allocate memory for input array
 31    float one_over_fftsize; // 1/fftsize
 32    std::complex<float> one_over_fftsize_c; // 1/fftsize complex value
 33 
 34  public:
 35    // Constructor for calculating multiple FFTs along a 1-d array
 36    fft(int nt, int n, int nblks, int in_ns, int out_ns, bool inv=false, std::complex<float>* const pin=NULL) {
 37        // Set class variables
 38        nthreads = nt;
 39        fftsize = n;
 40        one_over_fftsize = 1.0f/fftsize;
 41        one_over_fftsize_c =  std::complex<float>(one_over_fftsize, 0.0f);
 42        nblocks = nblks;
 43        in_nsteps = in_ns;
 44        out_nsteps = std::max(n, out_ns);
 45        U_D = 1;
 46        inverse = inv;
 47        ptr_in = pin;
 48        // Allocate input array
 49        if (ptr_in==NULL) 
 50            // a null ptr tells fft to allocate memory to input array
 51            in = (std::complex<float>*) fftwf_malloc(nblocks*std::max(in_nsteps,fftsize)*sizeof(std::complex<float>));
 52        else
 53            // a non-null ptr_in tells fft not to allocate memory to input array
 54            in = ptr_in;
 55        // Allocate output array
 56        out = (std::complex<float>*) fftwf_malloc(nblocks*out_nsteps*sizeof(std::complex<float>));
 57        // Initialize FFTW3 multi-threaded implementation
 58        fftwf_init_threads();
 59        fftwf_plan_with_nthreads(nthreads);
 60        // InstantiateForward FFT plan usinf FFTW3's advanced interface 
 61        plan = fftwf_plan_many_dft(1, &fftsize, nblocks, 
 62                                  reinterpret_cast<fftwf_complex *>(in), 
 63                                  NULL, 1, in_nsteps, 
 64                                  reinterpret_cast<fftwf_complex *>(out), 
 65                                  NULL, 1, out_nsteps, 
 66                                  inverse ? FFTW_BACKWARD : FFTW_FORWARD, 
 67                                  FFTW_MEASURE);
 68    }
 69    // Constructor calculating multiple FFTs along each row of a 2-d array
 70    // for use in polyphase filtering
 71    // Assume input array arranged with signals in uord rows 
 72    // in_ns and out_ns correspond to the number of samples along each row
 73    // nblks also indicates the number of FFT block per row
 74    fft(int nt, int n, int nblks, int in_ns, int out_ns, int uord, bool inv=false, std::complex<float>* const pin=NULL) {
 75        // Set class variables
 76        nthreads = nt;
 77        U_D = uord;
 78        fftsize = n;
 79        one_over_fftsize = 1.0f/fftsize;
 80        one_over_fftsize_c =  std::complex<float>(one_over_fftsize, 0.0f);
 81        nblocks = nblks;
 82        in_nsteps = in_ns;
 83        out_nsteps = std::max(n, out_ns);
 84        inverse = inv;
 85        ptr_in = pin;
 86        // Allocate input array
 87        if (ptr_in==NULL) 
 88            // a null ptr tells fft to allocate memory to input array
 89            in = (std::complex<float>*) fftwf_malloc(nblocks*U_D*std::max(in_nsteps,fftsize)*sizeof(std::complex<float>));
 90        else
 91            // a non-null ptr_in tells fft not to allocate memory to input array
 92            in = ptr_in;
 93        // Allocate output array
 94        out = (std::complex<float>*) fftwf_malloc(nblocks*U_D*out_nsteps*sizeof(std::complex<float>));
 95        // Calculate parameters for multi-threaded implementation
 96        // Initialize FFTW3 multi-threaded implementation
 97        fftwf_init_threads();
 98        fftwf_plan_with_nthreads(nthreads);
 99        // InstantiateForward FFT plan using FFTW3's guru interface
100        // Set up fft and howmany dimensions
101        fftwf_iodim fft_dim = {fftsize, 1, 1};
102        fftwf_iodim howmany_dim[2]; // 3d arrays for input and output
103        // This is to match the 1d array of input
104        howmany_dim[1] = {nblocks, in_nsteps, out_nsteps}; 
105        howmany_dim[0] = {U_D, nblocks*fftsize, nblocks*fftsize};
106        plan = fftwf_plan_guru_dft(1, &fft_dim, 2, howmany_dim,
107                                  reinterpret_cast<fftwf_complex *>(in), 
108                                  reinterpret_cast<fftwf_complex *>(out), 
109                                  inverse ? FFTW_BACKWARD : FFTW_FORWARD, 
110                                  FFTW_MEASURE);
111    }
112    // Destructor
113    ~fft(void) {
114        if (ptr_in==NULL) fftwf_free(in);
115        fftwf_free(out);
116        fftwf_destroy_plan(plan);
117    }
118    // Parameter getters
119    int get_fftsize(void) {
120        return fftsize;
121    }
122    int get_nblocks(void) {
123        return nblocks;
124    }
125    int get_in_nsteps() {
126        return in_nsteps;
127    }
128    int get_out_nsteps(void) {
129        return out_nsteps;
130    }
131    std::complex<float>* get_in(void) {
132        return in;
133    }
134    std::complex<float>* get_out(void) {
135        return out;
136    }
137    bool is_inverse(void) {
138        return inverse;
139    }
140    // Clean up
141    static void cleanup(void) {
142        fftwf_cleanup();
143        fftwf_cleanup_threads();
144    }
145    // Calculate FFT/IFFT
146    void calculate(void) {
147        // Call FFTW3 to Calculate FFT/IFFT
148        fftwf_execute(plan);
149        // If IFFT, divide by fftsize
150        if (inverse) {
151            for (int i=0; i<U_D*nblocks; i++) {
152                unsigned idx = i*out_nsteps;
153#ifdef USE_VOLK
154                volk_32fc_s32fc_multiply_32fc(out+idx, out+idx, one_over_fftsize_c, fftsize);
155#else
156                for (int j=0; j<fftsize; j++) {
157                    out[idx++] *= one_over_fftsize;
158                }
159#endif
160            }
161        }
162    }
163};
164#endif //FFT_H

  • The fft class defined above uses the FFTW3 advanced interface fftwf_plan_many_dft to create a plan to calculate multiple single-precision, 1-d FFTs along a 1-d array of samples and the guru interface fftwf_plan_guru_dft to create a plan to calculate multiple single-precision, 1-d FFTs along each row of a 2-d array of samples, using potentially multiple threads.

  • Both fftwf_plan_many_dft and fftwf_plan_guru_dft are actually more flexible that the way they are used here. Please refer to [2] for more usage information.

  • Can define the compile flag USE_VOLK to use the VOLK library for SIMD-accelerated arithmetic kernels such that volk_32fc_s32fc_multiply_32fc().

  • The internal multi-threading implementation of FFTW is used.

    Caution

    1. The internal multi-threading implementation of FFTW seems to interfere with setting the thread priority using the IHD method uhd::set_thread_priority_safe(). In particular, priority is set to the highest value regardless of what we set priority to, but realtime is not affected. I haven’t been able to work out the exact reason for that. FFTW3 uses Pthreads to set the priority of its threads and hence probably overwrites the value set by uhd::set_thread_priority_safe().

    2. Setting realtime to true actually results in slower FFT calculations in this case.

3.6.1.2. Usage#

  • 1-d constructor:

    fft(int nthreads, int fftsize, int nblocks, int in_nsteps, int out_nsteps, bool inverse=false, std::complex<float>* const in_ptr=NULL)

  • 2-d constructor:

    fft(int nthreads, int fftsize, int nblocks, int in_nsteps, int out_nsteps, int nrows, bool inverse=false, std::complex<float>* const in_ptr=NULL)

    • nthreads: number of threads to use

    • fftsize : size of FFT/IFFT

    • nblocks: number of FFTs/IFFTs to calculate along a row

    • in_nsteps: number of samples to step to the next block along a row in the input array

    • out_nsteps: number of samples to step to the next block along a row in the output array

    • nrows: number of rows in the 2-d array

    • inverse: FFT = false , IFFT = true (default = false)

    • in_ptr: pointer to the input array if non-NULL, internally allocate memory for the input array if NULL (default = NULL)

  • The restriction that out_nsteps must be at least as large as fftsize is imposed while in_nsteps may be smaller than fftsize.

  • The methods fft::get_in() and fft::get_out() can be used to obtain the pointers to the input and output arrays. The elements of both arrays should be of type std::complex<float>. The numbers of elements allocated for the input and output arrays are nblocks\(\cdot \max(\)fftsize\(,\) in_nsteps\()\) and nblocks\(\cdot \max(\)fftsize\(,\) out_nsteps\()\), respectively.

  • For nice termination, call the method fft::cleanup() before your main() exits.

  • For usage examples of the 1-d constructor, see test_fft.cpp.

  • For usage examples of the 2-d constructor, see filters.cpp.
