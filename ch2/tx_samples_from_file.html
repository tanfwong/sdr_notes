 21namespace po = boost::program_options;
 22
 23static bool stop_signal_called = false;
 24void sig_int_handler(int)
 25{
 26    stop_signal_called = true;
 27}
 28
 29template <typename samp_type>
 30void send_from_file(
 31    uhd::tx_streamer::sptr tx_stream, const std::string& file, size_t samps_per_buff)
 32{
 33    uhd::tx_metadata_t md;
 34    md.start_of_burst = false;
 35    md.end_of_burst   = false;
 36    std::vector<samp_type> buff(samps_per_buff);
 37    std::ifstream infile(file.c_str(), std::ifstream::binary);
 38
 39    // loop until the entire file has been read
 40
 41    while (not md.end_of_burst and not stop_signal_called) {
 42        infile.read((char*)&buff.front(), buff.size() * sizeof(samp_type));
 43        size_t num_tx_samps = size_t(infile.gcount() / sizeof(samp_type));
 44
 45        md.end_of_burst = infile.eof();
 46
 47        tx_stream->send(&buff.front(), num_tx_samps, md);
 48    }
 49
 50    infile.close();
 51}

 53int UHD_SAFE_MAIN(int argc, char* argv[])
 54{
 55    // variables to be set by po
 56    std::string args, file, type, ant, subdev, ref, wirefmt, channel;
 57    size_t spb;
 58    double rate, freq, gain, bw, delay, lo_offset;
 59
 60    // setup the program options
 61    po::options_description desc("Allowed options");
 62    // clang-format off
 63    desc.add_options()
 64        ("help", "help message")
 65        ("args", po::value<std::string>(&args)->default_value(""), "multi uhd device address args")
 66        ("file", po::value<std::string>(&file)->default_value("usrp_samples.dat"), "name of the file to read binary samples from")
 67        ("type", po::value<std::string>(&type)->default_value("short"), "sample type: double, float, or short")
 68        ("spb", po::value<size_t>(&spb)->default_value(10000), "samples per buffer")
 69        ("rate", po::value<double>(&rate), "rate of outgoing samples")
 70        ("freq", po::value<double>(&freq), "RF center frequency in Hz")
 71        ("lo-offset", po::value<double>(&lo_offset)->default_value(0.0),
 72            "Offset for frontend LO in Hz (optional)")
 73        ("lo_off", po::value<double>(&lo_offset),
 74            "(DEPRECATED) will go away soon! Use --lo-offset instead")
 75        ("gain", po::value<double>(&gain), "gain for the RF chain")
 76        ("ant", po::value<std::string>(&ant), "antenna selection")
 77        ("subdev", po::value<std::string>(&subdev), "subdevice specification")
 78        ("bw", po::value<double>(&bw), "analog frontend filter bandwidth in Hz")
 79        ("ref", po::value<std::string>(&ref)->default_value("internal"), "reference source (internal, external, mimo)")
 80        ("wirefmt", po::value<std::string>(&wirefmt)->default_value("sc16"), "wire format (sc8 or sc16)")
 81        ("delay", po::value<double>(&delay)->default_value(0.0), "specify a delay between repeated transmission of file (in seconds)")
 82        ("channel", po::value<std::string>(&channel)->default_value("0"), "which channel to use")
 83        ("repeat", "repeatedly transmit file")
 84        ("int-n", "tune USRP with integer-n tuning")
 85    ;
 86    // clang-format on
 87    po::variables_map vm;
 88    po::store(po::parse_command_line(argc, argv, desc), vm);
 89    po::notify(vm);
 90
 91    // print the help message
 92    if (vm.count("help")) {
 93        std::cout << boost::format("UHD TX samples from file %s") % desc << std::endl;
 94        return ~0;
 95    }
 96
 97    bool repeat = vm.count("repeat") > 0;
 98
 99    // create a usrp device
100    std::cout << std::endl;
101    std::cout << boost::format("Creating the usrp device with: %s...") % args
102              << std::endl;
103    uhd::usrp::multi_usrp::sptr usrp = uhd::usrp::multi_usrp::make(args);
104
105    // Lock mboard clocks
106    if (vm.count("ref")) {
107        usrp->set_clock_source(ref);
108    }
109
110    // always select the subdevice first, the channel mapping affects the other settings
111    if (vm.count("subdev"))
112        usrp->set_tx_subdev_spec(subdev);
113
114    std::cout << boost::format("Using Device: %s") % usrp->get_pp_string() << std::endl;
115
116    // set the sample rate
117    if (not vm.count("rate")) {
118        std::cerr << "Please specify the sample rate with --rate" << std::endl;
119        return ~0;
120    }
121    std::cout << boost::format("Setting TX Rate: %f Msps...") % (rate / 1e6) << std::endl;
122    usrp->set_tx_rate(rate);
123    std::cout << boost::format("Actual TX Rate: %f Msps...") % (usrp->get_tx_rate() / 1e6)
124              << std::endl
125              << std::endl;
126
127    // set the center frequency
128    if (not vm.count("freq")) {
129        std::cerr << "Please specify the center frequency with --freq" << std::endl;
130        return ~0;
131    }
132    std::cout << boost::format("Setting TX Freq: %f MHz...") % (freq / 1e6) << std::endl;
133    std::cout << boost::format("Setting TX LO Offset: %f MHz...") % (lo_offset / 1e6)
134              << std::endl;
135    uhd::tune_request_t tune_request;
136    tune_request = uhd::tune_request_t(freq, lo_offset);
137    if (vm.count("int-n"))
138        tune_request.args = uhd::device_addr_t("mode_n=integer");
139    usrp->set_tx_freq(tune_request);
140    std::cout << boost::format("Actual TX Freq: %f MHz...") % (usrp->get_tx_freq() / 1e6)
141              << std::endl
142              << std::endl;
143
144    // set the rf gain
145    if (vm.count("gain")) {
146        std::cout << boost::format("Setting TX Gain: %f dB...") % gain << std::endl;
147        usrp->set_tx_gain(gain);
148        std::cout << boost::format("Actual TX Gain: %f dB...") % usrp->get_tx_gain()
149                  << std::endl
150                  << std::endl;
151    }
152
153    // set the analog frontend filter bandwidth
154    if (vm.count("bw")) {
155        std::cout << boost::format("Setting TX Bandwidth: %f MHz...") % (bw / 1e6)
156                  << std::endl;
157        usrp->set_tx_bandwidth(bw);
158        std::cout << boost::format("Actual TX Bandwidth: %f MHz...")
159                         % (usrp->get_tx_bandwidth() / 1e6)
160                  << std::endl
161                  << std::endl;
162    }
163
164    // set the antenna
165    if (vm.count("ant"))
166        usrp->set_tx_antenna(ant);
167
168    // allow for some setup time:
169    std::this_thread::sleep_for(std::chrono::seconds(1));
170
171    // Check Ref and LO Lock detect
172    std::vector<std::string> sensor_names;
173    sensor_names = usrp->get_tx_sensor_names(0);
174    if (std::find(sensor_names.begin(), sensor_names.end(), "lo_locked")
175        != sensor_names.end()) {
176        uhd::sensor_value_t lo_locked = usrp->get_tx_sensor("lo_locked", 0);
177        std::cout << boost::format("Checking TX: %s ...") % lo_locked.to_pp_string()
178                  << std::endl;
179        UHD_ASSERT_THROW(lo_locked.to_bool());
180    }
181    sensor_names = usrp->get_mboard_sensor_names(0);
182    if ((ref == "mimo")
183        and (std::find(sensor_names.begin(), sensor_names.end(), "mimo_locked")
184                != sensor_names.end())) {
185        uhd::sensor_value_t mimo_locked = usrp->get_mboard_sensor("mimo_locked", 0);
186        std::cout << boost::format("Checking TX: %s ...") % mimo_locked.to_pp_string()
187                  << std::endl;
188        UHD_ASSERT_THROW(mimo_locked.to_bool());
189    }
190    if ((ref == "external")
191        and (std::find(sensor_names.begin(), sensor_names.end(), "ref_locked")
192                != sensor_names.end())) {
193        uhd::sensor_value_t ref_locked = usrp->get_mboard_sensor("ref_locked", 0);
194        std::cout << boost::format("Checking TX: %s ...") % ref_locked.to_pp_string()
195                  << std::endl;
196        UHD_ASSERT_THROW(ref_locked.to_bool());
197    }
198
199    // set sigint if user wants to receive
200    if (repeat) {
201        std::signal(SIGINT, &sig_int_handler);
202        std::cout << "Press Ctrl + C to stop streaming..." << std::endl;
203    }
204
205    // create a transmit streamer
206    std::string cpu_format;
207    std::vector<size_t> channel_nums;
208    if (type == "double")
209        cpu_format = "fc64";
210    else if (type == "float")
211        cpu_format = "fc32";
212    else if (type == "short")
213        cpu_format = "sc16";
214    uhd::stream_args_t stream_args(cpu_format, wirefmt);
215    channel_nums.push_back(boost::lexical_cast<size_t>(channel));
216    stream_args.channels             = channel_nums;
217    uhd::tx_streamer::sptr tx_stream = usrp->get_tx_stream(stream_args);
218
219    // send from file
220    do {
221        if (type == "double")
222            send_from_file<std::complex<double>>(tx_stream, file, spb);
223        else if (type == "float")
224            send_from_file<std::complex<float>>(tx_stream, file, spb);
225        else if (type == "short")
226            send_from_file<std::complex<short>>(tx_stream, file, spb);
227        else
228            throw std::runtime_error("Unknown type " + type);
229
230        if (repeat and delay > 0.0) {
231            std::this_thread::sleep_for(std::chrono::milliseconds(int64_t(delay * 1000)));
232        }
233    } while (repeat and not stop_signal_called);
234
235    // finished
236    std::cout << std::endl << "Done!" << std::endl << std::endl;
237
238    return EXIT_SUCCESS;
239}
