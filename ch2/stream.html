 24/*!
 25 * A struct of parameters to construct a streamer.
 26 *
 27 * Here is an example of how a stream args object could be used in conjunction
 28 * with uhd::device::get_rx_stream():
 29 *
 30 * \code{.cpp}
 31 * // 1. Create the stream args object and initialize the data formats to fc32 and sc16:
 32 * uhd::stream_args_t stream_args("fc32", "sc16");
 33 * // 2. Set the channel list, we want 3 streamers coming from channels
 34 * //    0, 1 and 2, in that order:
 35 * stream_args.channels = {0, 1, 2};
 36 * // 3. Set optional args:
 37 * stream_args.args["spp"] = "200"; // 200 samples per packet
 38 * // Now use these args to create an rx streamer:
 39 * // (We assume that usrp is a valid uhd::usrp::multi_usrp)
 40 * uhd::rx_streamer::sptr rx_stream = usrp->get_rx_stream(stream_args);
 41 * // Now, any calls to rx_stream must provide a vector of 3 buffers,
 42 * // one per channel.
 43 * \endcode
 44 *
 45 * \b Note: Not all combinations of CPU and OTW format have conversion support.
 46 * You may however write and register your own conversion routines.
 47 *
 48 * If you are creating stream args to connect to an RFNoC block, then you might
 49 * want to specify block ID and port, too:
 50 * \code{.cpp}
 51 * stream_args.args["block_id0"] = "0/Radio_0";
 52 * stream_args.args["block_id1"] = "0/Radio_1";
 53 * stream_args.args["block_id2"] = "0/Radio_1"; // Chan 1 and 2 go to the same radio
 54 * stream_args.args["block_port0"] = "0";
 55 * stream_args.args["block_port1"] = "0";
 56 * stream_args.args["block_port2"] = "1";
 57 * \endcode
 58 */
 59struct UHD_API stream_args_t
 60{
 61    //! Convenience constructor for streamer args
 62    stream_args_t(const std::string& cpu = "", const std::string& otw = "")
 63    {
 64        cpu_format = cpu;
 65        otw_format = otw;
 66    }
 67
 68    /*!
 69     * The CPU format is a string that describes the format of host memory.
 70     * Conversions for the following CPU formats have been implemented:
 71     *  - fc64 - complex<double>
 72     *  - fc32 - complex<float>
 73     *  - sc16 - complex<int16_t>
 74     *  - sc8 - complex<int8_t>
 75     *
 76     * The following are not implemented, but are listed to demonstrate naming convention:
 77     *  - f32 - float
 78     *  - f64 - double
 79     *  - s16 - int16_t
 80     *  - s8 - int8_t
 81     *
 82     * The CPU format can be chosen depending on what the application requires.
 83     */
 84    std::string cpu_format;
 85
 86    /*!
 87     * The OTW format is a string that describes the format over-the-wire.
 88     * The following over-the-wire formats have been implemented:
 89     *  - sc16 - Q16 I16
 90     *  - sc8 - Q8_1 I8_1 Q8_0 I8_0
 91     *  - sc12 (Only some devices)
 92     *
 93     * The following are not implemented, but are listed to demonstrate naming convention:
 94     *  - s16 - R16_1 R16_0
 95     *  - s8 - R8_3 R8_2 R8_1 R8_0
 96     *
 97     * Setting the OTW ("over-the-wire") format is, in theory, transparent to the
 98     * application, but changing this can have some side effects. Using less bits for
 99     * example (e.g. when going from `otw_format` `sc16` to `sc8`) will reduce the dynamic
100     * range, and increases quantization noise. On the other hand, it reduces the load on
101     * the data link and thus allows more bandwidth (a USRP N210 can work with 25 MHz
102     * bandwidth for 16-Bit complex samples, and 50 MHz for 8-Bit complex samples).
103     */
104    std::string otw_format;
105
106    /*!
107     * The args parameter is used to pass arbitrary key/value pairs.
108     * Possible keys used by args (depends on implementation):
109     *
110     * - fullscale: specifies the full-scale amplitude when using floats.
111     * By default, the fullscale amplitude under floating point is 1.0.
112     * Set the "fullscale" to scale the samples in the host to the
113     * expected input range and/or output range of your application.
114     *
115     * - peak: specifies a fractional sample level to calculate scaling with the sc8 wire
116     * format. When using sc8 samples over the wire, the device must scale samples (both
117     * on the host and in the device) to satisfy the dynamic range needs. The peak value
118     * specifies a fraction of the maximum sample level (1.0 = 100%). Set peak to
119     * max_sample_level/full_scale_level to ensure optimum dynamic range.
120     *
121     * - underflow_policy: how the TX DSP should recover from underflow.
122     * Possible options are "next_burst" or "next_packet".
123     * In the "next_burst" mode, the DSP drops incoming packets until a new burst is
124     * started. In the "next_packet" mode, the DSP starts transmitting again at the next
125     * packet.
126     *
127     * - spp: (samples per packet) controls the size of RX packets.
128     * When not specified, the packets are always maximum frame size.
129     * Users should specify this option to request smaller than default
130     * packets, probably with the intention of reducing packet latency.
131     *
132     * - noclear: Used by tx_dsp_core_200 and rx_dsp_core_200
133     *
134     * The following are not implemented, but are listed for conceptual purposes:
135     * - function: magnitude or phase/magnitude
136     * - units: numeric units like counts or dBm
137     *
138     * Other options are device-specific:
139     * - port, addr: Alternative receiver streamer destination.
140     */
141    device_addr_t args;
142
143    /*!
144     * The channels is a list of channel numbers.
145     * Leave this blank to default to channel 0 (single-channel application).
146     * Set channels for a multi-channel application.
147     * Channel mapping depends on the front-end selection (see also \ref config_subdev).
148     *
149     * A very simple example is an X300 with two daughterboards and a subdev spec
150     * of `A:0 B:0`. This means the device has two channels available.
151     *
152     * Setting `stream_args.channels = (0, 1)` therefore configures MIMO streaming
153     * from both channels. By switching the channel indexes, `stream_args.channels = (1,
154     * 0)`, the channels are switched and the first channel of the USRP is mapped to the
155     * second channel in the application.
156     *
157     * If only a single channel is used for streaming, `stream_args.channels = (1,)` would
158     * only select a single channel (in this case, the second one). When streaming
159     * a single channel from the B-side radio of a USRP, this is a more versatile solution
160     * than setting the subdev globally to "B:0".
161     */
162    std::vector<size_t> channels;
163};

165/*!
166 * The RX streamer is the host interface to receiving samples.
167 * It represents the layer between the samples on the host
168 * and samples inside the device's receive DSP processing.
169 */
170class UHD_API rx_streamer : uhd::noncopyable
171{
172public:
173    typedef boost::shared_ptr<rx_streamer> sptr;
174
175    virtual ~rx_streamer(void);
176
177    //! Get the number of channels associated with this streamer
178    virtual size_t get_num_channels(void) const = 0;
179
180    //! Get the max number of samples per buffer per packet
181    virtual size_t get_max_num_samps(void) const = 0;
182
183    //! Typedef for a pointer to a single, or a collection of recv buffers
184    typedef ref_vector<void*> buffs_type;
185
186    /*!
187     * Receive buffers containing samples described by the metadata.
188     *
189     * Receive handles fragmentation as follows:
190     * If the buffer has insufficient space to hold all samples
191     * that were received in a single packet over-the-wire,
192     * then the buffer will be completely filled and the implementation
193     * will hold a pointer into the remaining portion of the packet.
194     * Subsequent calls will load from the remainder of the packet,
195     * and will flag the metadata to show that this is a fragment.
196     * The next call to receive, after the remainder becomes exhausted,
197     * will perform an over-the-wire receive as usual.
198     * See the rx metadata fragment flags and offset fields for details.
199     *
200     * This is a blocking call and will not return until the number
201     * of samples returned have been written into each buffer.
202     * Under a timeout condition, the number of samples returned
203     * may be less than the number of samples specified.
204     *
205     * The one_packet option allows the user to guarantee that
206     * the call will return after a single packet has been processed.
207     * This may be useful to maintain packet boundaries in some cases.
208     *
209     * Note on threading: recv() is *not* thread-safe, to avoid locking
210     * overhead. The application calling recv() is responsible for making
211     * sure that not more than one thread can call recv() on the same streamer
212     * at the same time. If there are multiple streamers, receiving from
213     * different sources, then those may be called from different threads
214     * simultaneously.
215     *
216     * \param buffs a vector of writable memory to fill with samples
217     * \param nsamps_per_buff the size of each buffer in number of samples
218     * \param metadata data to fill describing the buffer
219     * \param timeout the timeout in seconds to wait for a packet
220     * \param one_packet return after the first packet is received
221     * \return the number of samples received or 0 on error
222     */
223    virtual size_t recv(const buffs_type& buffs,
224        const size_t nsamps_per_buff,
225        rx_metadata_t& metadata,
226        const double timeout  = 0.1,
227        const bool one_packet = false) = 0;
228
229    /*!
230     * Issue a stream command to the usrp device.
231     * This tells the usrp to send samples into the host.
232     * See the documentation for stream_cmd_t for more info.
233     *
234     * With multiple devices, the first stream command in a chain of commands
235     * should have a time spec in the near future and stream_now = false;
236     * to ensure that the packets can be aligned by their time specs.
237     *
238     * \param stream_cmd the stream command to issue
239     */
240    virtual void issue_stream_cmd(const stream_cmd_t& stream_cmd) = 0;
241};

243/*!
244 * The TX streamer is the host interface to transmitting samples.
245 * It represents the layer between the samples on the host
246 * and samples inside the device's transmit DSP processing.
247 */
248class UHD_API tx_streamer : uhd::noncopyable
249{
250public:
251    typedef boost::shared_ptr<tx_streamer> sptr;
252
253    virtual ~tx_streamer(void);
254
255    //! Get the number of channels associated with this streamer
256    virtual size_t get_num_channels(void) const = 0;
257
258    //! Get the max number of samples per buffer per packet
259    virtual size_t get_max_num_samps(void) const = 0;
260
261    //! Typedef for a pointer to a single, or a collection of send buffers
262    typedef ref_vector<const void*> buffs_type;
263
264    /*!
265     * Send buffers containing samples described by the metadata.
266     *
267     * Send handles fragmentation as follows:
268     * If the buffer has more items than the maximum per packet,
269     * the send method will fragment the samples across several packets.
270     * Send will respect the burst flags when fragmenting to ensure
271     * that start of burst can only be set on the first fragment and
272     * that end of burst can only be set on the final fragment.
273     *
274     * This is a blocking call and will not return until the number
275     * of samples returned have been read out of each buffer.
276     * Under a timeout condition, the number of samples returned
277     * may be less than the number of samples specified.
278     *
279     * Note on threading: send() is *not* thread-safe, to avoid locking
280     * overhead. The application calling send() is responsible for making
281     * sure that not more than one thread can call send() on the same streamer
282     * at the same time. If there are multiple streamers, transmitting to
283     * different destinations, then those may be called from different threads
284     * simultaneously.
285     *
286     * \param buffs a vector of read-only memory containing samples
287     * \param nsamps_per_buff the number of samples to send, per buffer
288     * \param metadata data describing the buffer's contents
289     * \param timeout the timeout in seconds to wait on a packet
290     * \return the number of samples sent
291     */
292    virtual size_t send(const buffs_type& buffs,
293        const size_t nsamps_per_buff,
294        const tx_metadata_t& metadata,
295        const double timeout = 0.1) = 0;
296
297    /*!
298     * Receive and asynchronous message from this TX stream.
299     * \param async_metadata the metadata to be filled in
300     * \param timeout the timeout in seconds to wait for a message
301     * \return true when the async_metadata is valid, false for timeout
302     */
303    virtual bool recv_async_msg(
304        async_metadata_t& async_metadata, double timeout = 0.1) = 0;
305};
